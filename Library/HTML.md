HTML

<!DOCTYPE html> — тип документа HTML5
<html></html> — вся страница
<head></head> — служебная часть
<title></title> — название вкладки
<link rel="stylesheet" href="file.css"> — подключение CSS
<meta name="viewport" content="width=device-width, initial-scale=1.0">
<script></script>
<body></body> — всё, что видно на странице



Текст

<h1>–<h6> — заголовки
<p> — абзац
<br> — перенос строки
<hr> — линия
<b> — жирный
<i> — курсив
<u> — подчёркнутый
<div> — блочный контейнер
<!-- --> — комментарий



Списки

<ul> — маркированный список
<ol> — нумерованный список
<li> — элемент списка



Ссылки

<a href=""> — ссылка
target="_blank" — открыть в новой вкладке
mailto: — письмо
tel: — звонок
href="#" — заглушка
href="page.html" — другая страница
href="folder/page.html" — другая страница в другой папке



Изображения

<img src="" alt=""> — изображение
draggable="false" — запрет перетаскивания


Таблицы

<table> — таблица
<caption> — заголовок таблицы
<thead> — шапка таблицы
<tbody> — основная часть
<tfoot> — нижняя часть
<tr> — строка
<th> — заголовок столбца
<td> — ячейка


Формы

<form> — форма
method="GET" — данные в URL
method="POST" — данные скрыты
<label> — подпись поля
<input> — поле ввода
<textarea> — большое поле
<select> — список
<option> — вариант
<button> — кнопка


Типы input

type="text" — текст
type="password" — пароль
type="email" — email
type="number" — число
type="date" — дата
type="checkbox" — галочка
type="radio" — выбор одного
